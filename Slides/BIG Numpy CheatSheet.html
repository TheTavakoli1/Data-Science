import numpy as np # ─── نرمالیزاسیون HU به [0,1] ─────────────────────── def normalize_ct(volume, hu_min=-1000, hu_max=400): volume = np.clip(volume, hu_min, hu_max) return (volume - hu_min) / (hu_max - hu_min) # ─── Window/Level (Lung Window) ───────────────────── def apply_window(img, center=-600, width=1500): low = center - width / 2 high = center + width / 2 return np.clip(img, low, high) # ─── Extract 3D patches برای nodule detection ──────── def extract_patch_3d(volume, center, size=64): z, y, x = center h = size // 2 return volume[z-h:z+h, y-h:y+h, x-h:x+h] # ─── Binary mask و اعمال به حجم ────────────────────── mask = (ct_volume > -500) & (ct_volume < -100) # ریه lung = ct_volume * mask # ─── آمار هر نودول ─────────────────────────────────── nodule_region = ct_volume[mask] stats = { 'mean': np.mean(nodule_region), 'std': np.std(nodule_region), 'p10': np.percentile(nodule_region, 10), 'p90': np.percentile(nodule_region, 90), 'volume_voxels': np.sum(mask) } # ─── Data augmentation با flip/rotate ──────────────── aug_flip_h = np.flip(patch, axis=2) # horizontal aug_flip_v = np.flip(patch, axis=1) # vertical aug_rot90 = np.rot90(patch, k=1, axes=(1,2)) # rotate slice # ─── Batch برای مدل ────────────────────────────────── batch = np.stack([p1, p2, p3], axis=0) # (N, D, H, W) batch = batch[:, np.newaxis, ...] # (N,1,D,H,W) for PyTorch batch_f32 = batch.astype(np.float32) / 255.0

Performance Tips — بهینه‌سازی

نکات کلیدی بهینه‌سازی

# ✗ کُند — حلقه Python result = [x**2 + 1 for x in arr] # ✓ سریع — vectorized result = arr**2 + 1 # ✓ بدون allocation اضافه np.add(a, b, out=c) # ✓ Memory-mapped برای CT volume بزرگ volume = np.memmap('ct_scan.dat', dtype='float32', mode='r', shape=(512,512,300)) # ✓ einsum — ضرب batch ماتریس‌ها # a: (B,M,K) b: (B,K,N) → (B,M,N) result = np.einsum('bmk,bkn->bmn', a, b)
